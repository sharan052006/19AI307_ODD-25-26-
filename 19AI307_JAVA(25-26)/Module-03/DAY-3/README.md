import java.util.*;

abstract class SignalAgent {
    abstract int decodeSignal(int[] signal);
}

class PrimeAgent extends SignalAgent {
    boolean isPrime(int n) {
        if (n <= 1) return false;
        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0)
                return false;
        }
        return true;
    }

    @Override
    int decodeSignal(int[] signal) {
        int sum = 0;
        for (int num : signal) {
            if (isPrime(num))
                sum += num;
        }
        return sum;
    }
}

class MirrorAgent extends SignalAgent {
    @Override
    int decodeSignal(int[] signal) {
        int n = signal.length;
        for (int i = 0; i < n / 2; i++) {
            if (signal[i] != signal[n - 1 - i]) {
             
                return -1;
            }
        }
        return 1;
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] signal = new int[n];
        for (int i = 0; i < n; i++)
            signal[i] = sc.nextInt();
        int type = sc.nextInt();
        
        SignalAgent agent;
        
        if (type == 1) {
            agent = new PrimeAgent();
            System.out.println(agent.decodeSignal(signal));
        } else if (type == 2) {
            agent = new MirrorAgent();
            int result = agent.decodeSignal(signal);
            if (result == 1)
                System.out.println("BALANCED");
            else
                System.out.println("BROKEN");
        }
    }
}
